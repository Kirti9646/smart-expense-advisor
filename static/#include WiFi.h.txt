#include <WiFi.h>
#include <WebServer.h>

#define RELAY_PIN 23   // change GPIO if needed

const char* ssid = "ESP32_Relay";
const char* password = "12345678";

WebServer server(80);

bool relayState = false;

// ---------- WEB PAGE ----------
String getPage() {
  String page = "<!DOCTYPE html><html><head>";
  page += "<title>ESP32 Relay</title>";
  page += "<meta name='viewport' content='width=device-width, initial-scale=1'>";
  page += "<style>";
  page += "body{text-align:center;font-family:Arial;}";
  page += "button{padding:20px;font-size:20px;margin:10px;}";
  page += "</style></head><body>";

  page += "<h2>ESP32 Relay Control</h2>";

  if(relayState)
    page += "<h3>STATUS: ON</h3>";
  else
    page += "<h3>STATUS: OFF</h3>";

  page += "<a href='/on'><button>ON</button></a>";
  page += "<a href='/off'><button>OFF</button></a>";

  page += "</body></html>";
  return page;
}

// ---------- HANDLERS ----------
void handleRoot() {
  server.send(200, "text/html", getPage());
}

void handleON() {
  digitalWrite(RELAY_PIN, LOW);   // LOW trigger relay
  relayState = true;
  server.send(200, "text/html", getPage());
}

void handleOFF() {
  digitalWrite(RELAY_PIN, HIGH);
  relayState = false;
  server.send(200, "text/html", getPage());
}

// ---------- SETUP ----------
void setup() {
  Serial.begin(115200);

  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, HIGH);   // relay OFF initially

  WiFi.mode(WIFI_AP);
  WiFi.softAP(ssid, password);

  Serial.println("Hotspot Started");
  Serial.print("Open Browser → http://");
  Serial.println(WiFi.softAPIP());

  server.on("/", handleRoot);
  server.on("/on", handleON);
  server.on("/off", handleOFF);

  server.begin();
}

// ---------- LOOP ----------
void loop() {
  server.handleClient();
}
